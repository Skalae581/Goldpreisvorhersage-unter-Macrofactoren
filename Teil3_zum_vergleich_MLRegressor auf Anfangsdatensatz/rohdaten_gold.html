# Bibliotheken importieren
import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import MinMaxScaler
from sklearn.neural_network import MLPRegressor
from sklearn.impute import SimpleImputer
from sklearn.metrics import mean_squared_error, mean_absolute_error, r2_score
import matplotlib.pyplot as plt

# Schritt 1: Daten laden
df = pd.read_csv("FINAL_USO.csv")

# Schritt 2: Datum als Index setzen
df['Date'] = pd.to_datetime(df['Date'])
df.set_index('Date', inplace=True)

# Schritt 3: Zielvariable definieren (Tagesrendite: Close - Open)
#df['Gold_Return'] = df['Close'] - df['Open']
# Gold_Return so definieren, dass es die Rendite von morgen ist, basierend auf den heutigen Daten
df['Gold_Return'] = (df['Close'].shift(-1) - df['Open'].shift(-1)).fillna(method='ffill')
# Schritt 4: Features & Zieltrennung
X = df.drop(columns=['Gold_Return'])
y = df['Gold_Return']

# Schritt 5: Fehlende Werte ersetzen (Imputation)
X_imputed = SimpleImputer(strategy='mean').fit_transform(X)
y_imputed = SimpleImputer(strategy='mean').fit_transform(y.values.reshape(-1, 1)).ravel()

# Schritt 6: Normalisierung der Eingabedaten
scaler = MinMaxScaler()
X_scaled = scaler.fit_transform(X_imputed)

# Schritt 7: Train-Test-Split
X_train, X_test, y_train, y_test = train_test_split(X_scaled, y_imputed, test_size=0.2, random_state=42)

# Schritt 8: MLP-Modell erstellen & trainieren
mlp = MLPRegressor(hidden_layer_sizes=(100, 50), max_iter=1000, random_state=42)
mlp.fit(X_train, y_train)

# Schritt 9: Vorhersagen & Bewertung
y_pred = mlp.predict(X_test)

# Metriken berechnen
r2 = r2_score(y_test, y_pred)
rmse = np.sqrt(mean_squared_error(y_test, y_pred))
mae = mean_absolute_error(y_test, y_pred)

print(f"R: {r2:.4f}, RMSE: {rmse:.4f}, MAE: {mae:.4f}")

# Schritt 10: Plot  Vergleich echte vs. vorhergesagte Werte
plt.figure(figsize=(12, 4))
plt.plot(y_test[:100], label='Echt', color='blue')
plt.plot(y_pred[:100], label='Vorhersage', color='orange', linestyle='--')
plt.title("Tagesrendite Gold: Echt vs. Vorhergesagt (erste 100 Beobachtungen)")
plt.xlabel("Beobachtung")
plt.ylabel("Rendite")
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.savefig("Tagesrendite Gold: Echt vs. Vorhergesagt (erste 100 Beobachtungen).png", dpi=300)
plt.show()

import pandas as pd

# Daten laden

# Zielvariable: Tagesrendite
df['Gold_Return'] = df['Close'] - df['Open']

# Plot 1: Preisverlauf
plt.figure(figsize=(14, 5))
plt.plot(df.index, df['Close'], label='Close')
plt.plot(df.index, df['Open'], label='Open', linestyle='--')
plt.title('Goldpreis Verlauf')
plt.xlabel('Datum')
plt.ylabel('Preis')
plt.legend()
plt.grid(True)
plt.tight_layout()

# Fehlende Werte ersetzen

# MLP-Modell
