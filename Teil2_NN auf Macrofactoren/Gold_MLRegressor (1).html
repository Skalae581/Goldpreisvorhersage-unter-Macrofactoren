def load_and_prepare_safe(file_path, date_format=None):
    if not os.path.exists(file_path):
        print(f"❌ Datei nicht gefunden: {file_path}")
        return None
    try:
        df=pd.read_csv(file_path)
        print(f"✔️ {len(df)} Zeilen erfolgreich geladen und vorbereitet.")
        return df
    except Exception as e:
        print(f"❗ Fehler beim Laden/Verarbeiten der Datei: {e}")
        return None

from sklearn.metrics import mean_absolute_error, root_mean_squared_error, r2_score
from sklearn.model_selection import cross_val_score, KFold
from sklearn.model_selection import cross_validate, TimeSeriesSplit
import numpy as np

# Bestes Modell extrahieren
best_model = grid_search.best_estimator_

# --- Bewertung auf Trainingsdaten mit Cross-Validation ---
cv = KFold(n_splits=5, shuffle=True, random_state=42)
#cv = TimeSeriesSplit(n_splits=5)

# RMSE
rmse_scores=cross_val_score(best_model,X_train_scaled,y_train_imputed,cv=cv,scoring='neg_root_mean_squared_error')
rmse_scores_pos=-rmse_scores

# R²
r2_scores=cross_val_score(best_model, X_train_scaled, y_train_imputed,cv=cv,scoring='r2')

# MAE
mae_scores=cross_val_score(best_model, X_train_scaled, y_train_imputed,cv=cv, scoring='neg_mean_absolute_error')
mae_scores_pos=-mae_scores

# --- Ausgabe Training (Validierung) ---
print("📊 Cross-Validation (Trainingsdaten):")
print(f"RMSE pro Fold: {rmse_scores_pos}")
print(f"Durchschnittlicher RMSE: {np.mean(rmse_scores_pos):.4f}")
print(f"Durchschnittlicher R²:   {np.mean(r2_scores):.4f}")
print(f"Durchschnittlicher MAE:  {np.mean(mae_scores_pos):.4f}")

y_test_pred = best_model.predict(X_test_scaled)
